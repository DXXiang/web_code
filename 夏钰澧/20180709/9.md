	public static boolean isPalindrome(int x) {
		if(x<0)
			return false;
		ArrayList<Integer> nums=new ArrayList();
		while(x!=0) {
			int count =x%10;
			x/=10;
			nums.add(count);
		}
		int j=nums.size()-1;
		for(int i=0;i<j;i++,j--) {
			if(nums.get(i)!=nums.get(j))
				return false;
		}
		return true;
	   }
}
