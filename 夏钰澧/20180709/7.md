	//反转整数
	public static void main(String args[]) {
		System.out.println(reverse(123));
	}
	  public static int reverse(int x) {
	        int tem = 0;
	        while (x != 0) {
	            int count = x % 10;
	            x /= 10;
	            if (tem > Integer.MAX_VALUE || tem<Integer.MIN_VALUE) {
	            	return 0;
	            } 
	           tem = tem * 10 + count;
	        }
	        return tem;
	    }
}
