class Solution {
    public int[] intersection(int[] nums1, int[] nums2) {
        int[]no={};
        if((nums1.length==0)||(nums2.length==0))    
            return no;        
        int[]n;
        n=new int[nums1.length];      
        int l=0;        
        for(int i=0;i<nums1.length;i++)
        {
            for(int j=0;j<nums2.length;j++)
            { 
                if(nums2[j]==nums1[i])
                {
                     boolean add=true;
                     for(int u=0;u<l;u++)
                 {
                    if(nums2[j]==n[u]) 
                     add=false;                 
                 }
                if(add) 
                 {
                    n[l]=nums2[j];
                     l++;
                 }   
                }
            }
        }  
        int[]p;
        p=new int[l];      
        for(int q=0;q<l;q++)
        {
           p[q]=n[q];         
        }     
        return p;
    }
}
